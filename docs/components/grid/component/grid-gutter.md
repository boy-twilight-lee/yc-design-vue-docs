### 区块间隔

通过在 <yc-tag>Row</yc-tag> 上指定 <yc-tag>Gutter</yc-tag> 可以增加栅格的区域间隔。

<div class="cell-demo vp-raw">
 <div>
    <p>Horizontal</p>
    <yc-row class="grid-demo" :gutter="24">
      <yc-col :span="12">
        <div>col - 12</div>
      </yc-col>
      <yc-col :span="12">
        <div>col - 12</div>
      </yc-col>
    </yc-row>
    <p>Responsive</p>
    <yc-row class="grid-demo" :gutter="{ md: 8, lg: 24, xl: 32 }">
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
    </yc-row>
    <p>Horizontal and Vertical</p>
    <yc-row class="grid-demo" :gutter="[24, 12]">
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
      <yc-col :span="6">
        <div>col - 6</div>
      </yc-col>
    </yc-row>
  </div>
</div>

  display: flex;
